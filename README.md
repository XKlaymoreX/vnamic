## Install

```bash
npm install --save tvnamic
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'tvnamic'
import 'tvnamic/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
